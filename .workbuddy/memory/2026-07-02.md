### 技术细节

#### 批处理脚本内容
```batch
@echo off
chcp 65001 > nul
echo Updating Quotix...

REM 等待当前应用关闭
timeout /t 5 /nobreak > nul

REM 运行安装程序（静默安装）
echo Installing update...
"C:\Users\Evans\AppData\Local\Quotix\Updates\Quotix_Setup_1.0.4.exe" /SILENT

REM 等待安装完成
timeout /t 10 /nobreak > nul

REM 删除安装包
echo Deleting installer...
del "C:\Users\Evans\AppData\Local\Quotix\Updates\Quotix_Setup_1.0.4.exe" /Q

REM 删除此批处理脚本
del "%~f0" /Q

REM 启动新版本的应用
echo Starting Quotix...
start "" "C:\Users\Evans\AppData\Local\Programs\Quotix\Launcher\Quotix.exe"

exit
```
